pythonimport tensorflow as tf
print(f"TensorFlow version: {tf.__version__}")

# Check GPU/DML devices
gpu_devices = tf.config.list_physical_devices('GPU')
dml_devices = tf.config.list_physical_devices('DML')

print(f"GPU devices: {gpu_devices}")
print(f"DML devices: {dml_devices}")

if gpu_devices or dml_devices:
    print("‚úÖ AMD GPU detected!")
    
    # Test simple computation on GPU
    try:
        with tf.device('/GPU:0'):
            a = tf.constant([[1.0, 2.0], [3.0, 4.0]])
            b = tf.constant([[2.0, 1.0], [1.0, 2.0]])
            c = tf.matmul(a, b)
            print("üöÄ GPU computation test:")
            print(c.numpy())
    except Exception as e:
        print(f"GPU test error: {e}")
else:
    print("‚ö†Ô∏è No GPU detected")
